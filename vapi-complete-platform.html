class CallManager {
    let vapi = VoicePartnerAI(
        apiKey: "your-api-key",
        serverUrl: "https://api.voicepartnerai.eu"
    )
    
    func startCall() {
        vapi.start(
            assistantId: "assistant-id",
            customerNumber: "+49301234567"
        ) { result in
            switch result {
            case .success(let call):
                print("Call started: \(call.id)")
            case .failure(let error):
                print("Error: \(error)")
            }
        }
    }
}

import eu.voicepartnerai.sdk.VoicePartnerAI

class CallManager {
    private val vapi = VoicePartnerAI(
        apiKey = "your-api-key",
        serverUrl = "https://api.voicepartnerai.eu"
    )
    
    suspend fun startCall() {
        try {
            val call = vapi.calls.create(
                assistantId = "assistant-id",
                customerNumber = "+49301234567"
            )
            Log.d("VoicePartnerAI", "Call started: ${call.id}")
        } catch (e: Exception) {
            Log.e("VoicePartnerAI", "Error starting call", e)
        }
    }
}
